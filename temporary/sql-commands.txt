CREATE TABLE TriangleData(
    item_level int,
    item_position int,
    item_value int,
    PRIMARY KEY(item_level, item_position)
);

CREATE TABLE PathData(
    path_outer_id int,
    path_inner_id int,
    direction varchar(5),
    PRIMARY KEY(path_outer_id, path_inner_id)
);

CREATE TABLE Node(
    node_id int AUTO_INCREMENT,
    node_value int,
    max_sum int,
    max_path_outer_id int,
    max_path_inner_id int,
    execution_time float,
    max_direction varchar(5),
    parent_id int,
    PRIMARY KEY(node_id)
);

ALTER TABLE Node ADD CONSTRAINT fk_1
FOREIGN KEY(max_path_outer_id) REFERENCES PathData(path_outer_id);

ALTER TABLE Node ADD CONSTRAINT fk_2
FOREIGN KEY(max_path_inner_id) REFERENCES PathData(path_inner_id);

ALTER TABLE Node ADD CONSTRAINT fk_3
FOREIGN KEY(parent_id) REFERENCES Node(node_id);

CREATE TABLE Triangle(
    triangle_id int AUTO_INCREMENT,
    data_level int,
    data_position int,
    root_id int,
    PRIMARY KEY(triangle_id),
    FOREIGN KEY(data_level) REFERENCES TriangleData(item_level),
    FOREIGN KEY(data_position) REFERENCES TriangleData(item_position),
    FOREIGN KEY(root_id) REFERENCES Node(node_id)
);
